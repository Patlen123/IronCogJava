	id = modpolitical.3

	desc = modpolitical.3.d
	picture = GFX_report_event_fascist_gathering
	
	trigger = {
		has_idea_with_trait = autocratic_advocate
		NOT = { has_government = neutrality }
		has_country_flag = patcrat_cons_support
		neutrality > 0.3
		is_puppet = no
	}

	mean_time_to_happen = {
		days = 200
		modifier = {
			factor = 0.5
			neutrality > 0.4
		}
	}

	immediate = {
		hidden_effect = { remove_ideas = traditionalist_clubs }
	}
	
	option = {
		name = modpolitical.3.a
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}

# Civil War for neutrality
country_event = {
	id = modpolitical.4
	title = modpolitical.4.t
	desc = modpolitical.4.d
	picture = GFX_report_event_fascist_gathering
	
	trigger = {
		has_idea_with_trait = autocratic_advocate
		NOT = { 
			has_government = neutrality
			is_subject = yes
		}
		neutrality > 0.25
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.75
			has_country_flag = patcrat_youth_support
		}
	}
	
	option = {
		name = modpolitical.4.a
		if = {
			limit = {
				has_government = democratic
			}
			if = {
				limit = { neutrality > 0.4 }
				start_civil_war = {
					ruling_party = neutrality
					ideology = democratic
					size = 0.6
				}
			}
			if = {
				limit = { NOT = { neutrality > 0.4 } }
				start_civil_war = {
					ruling_party = neutrality
					ideology = democratic
					size = 0.7
				}
			}
		}
		if = {
			limit = {
				has_government = communism
			}
			if = {
				limit = { neutrality > 0.4 }
				start_civil_war = {
					ruling_party = neutrality
					ideology = communism
					size = 0.6
				}
			}
			if = {
				limit = {
					has_government = fascism
				}
				start_civil_war = {
					ruling_party = neutrality
					ideology = fascism
					size = 0.7
				}
			}
		}
		remove_ideas = traditionalist_clubs
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
	option = {
		name = modpolitical.4.b
	}
}
