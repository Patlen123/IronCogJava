country_event = {
	id = modcolonies.1
	title = modcolonies.1.t
	desc = modcolonies.1.d
	picture = GFX_report_event_german_troops
	fire_only_once = yes
	trigger = {
		all_country = {
			has_idea = GER_OST
		}
	}
	mean_time_to_happen = { days = 2 }

			remove_ideas = GER_OST
			add_timed_idea = {
				idea = GER_OST_2
				days = 365
			}
	}
}
