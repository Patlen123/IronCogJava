# England Joins Comintern
news_event = {
	id = modnews.13
	title = modnews.13.t
	desc = modnews.13.d
	picture = GFX_news_event_french_parade

	major = yes

	is_triggered_only = yes

	immediate = {
		ENG = {
			save_event_target_as = England		
		}
	}

	#United towards the World Revolution!
	option = {
		name = modnews.13.a
		trigger = {
			OR = {
				has_government = communism
				tag = ENG
			}
		}
	}

	#Their alliance will crumble soon enough.
	option = {
		name = modnews.13.b
		trigger = {
			has_government = fascism
			NOT = {
				tag = ENG
			}
		}
	}

	#The enemies of democracy must not be allowed to grow stronger.
	option = {
		name = modnews.13.c
		trigger = {
			has_government = democratic
			NOT = {
				tag = ENG
			}
		}
	}

	#England has picked its side.
	option = {
		name = modnews.13.d
		trigger = {
			has_government = neutrality
			NOT = {
				tag = ENG
			}
		}
	}
