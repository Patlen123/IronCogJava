#Move cap to Constantinople
country_event = {
	id = modturkey.2
	title = modturkey.2.t
	desc = modturkey.2.d
	picture = GFX_event_ottoman
	fire_only_once = yes
	trigger = {
		tag = TUR
		has_cosmetic_tag = OTT
		owns_state = 341
	}
	mean_time_to_happen = { days = 2 }
	option = {
		name = modturkey.2.a
		set_capital = 341
		set_province_name = { id = 9833 name = "Konstantiniyye"}
	}
	option = {
		name = modturkey.2.b
	}
}
